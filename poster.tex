    \innerblock[]{Theorem}{
      Consider the following causal target parameter,
      \begin{align*}
        \Psi^F(P_{U,X,0}) &= \int_{w,z} \E[Y(1,z) - Y(0,z) \mid W=w] \\
                          & \qquad\qquad\qquad p_Z(z \mid A=0,w)p_W(w)~dz~dw\;.
      \end{align*}
      Under the assumptions above, the corresponding statistical estimand may
      be expressed
      \begin{align*}
        \Psi(P) &= \int_w \int_z \{\overline{Q}_Y(w,A=1,z) -
        \overline{Q}_Y(w,A=0,z)\} \\
                & \qquad\qquad\qquad p_Z(z \mid A=0,w)~dz~p_W(w)~dw\\
                &= \E_P\E_P \{\E_P(Y \mid W,A=1,Z) \\
                &\qquad\qquad\qquad - \E_P(Y \mid W,A=0,Z) \mid A=0,W\} \ .
      \end{align*}
    }
